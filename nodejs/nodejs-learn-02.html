    # 2、mysql.js

    var mysql = require('mysql');
var config = require('./db');
var pool = mysql.createPool({
    host: config.database.HOST,
    user: config.database.USERNAME,
    password: config.database.PASSWORD,
    database: config.database.DATABASE
});

class Mysql{
    constructor() {
    }
    query(){
        return new Promise((resolve, reject)=>{
            pool.query('SELECT * FROM ceshidata',function(error,results,fields){
                if(error){
                    throw error;
                };
                resolve(results);
            });
        })
    }
    insert(){
        return new Promise((resolve, reject)=>{
            var insert_sql = 'insert into ceshidata (name,sex) values('+'\'ll\','+'\'女\')';
            pool.query(insert_sql,function(error,results,fields){
                if(error){
                    throw error;
                };
                resolve(results);
            });
        })
    }
    update(){
        return new Promise((resolve,reject)=>{
            var update_sql = 'update ceshidata set sex = ' + '\'none\' where name = '+ '\'ll\'';
            pool.query(update_sql,function (error,results,fields){
                if(error) {
                    throw error;
                } ;
                resolve(results);
            })
        });
    }

    delete(){
        return new Promise((resolve,reject)=>{
            var delete_sql = 'delete from ceshidata where name = ' + '\'ll\'';
            pool.query(delete_sql,function(error,results,fields){
                if(error)   throw error;
                resolve(results);
            })

        })
    }
}

module.exports = new Mysql()

    # 3、index.js

    const Koa = require('koa');
const config = require('./db');
const mysql = require('./mysql');

const app = new Koa();
app.use(async (ctx)=>{
    //let a = mysql.insert();
    //let b = mysql.update();
    //let c = mysql.delete();
    let data = await mysql.query();
    ctx.body={
        "code": 1,
        "data": data,
        "mesg": 'ok'
    }
});
app.listen(config.port);
console.log(`listen on port ${config.port}`);
